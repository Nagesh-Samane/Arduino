
int  led2=9,led1=5,count=1,led3=8, led4=4;
void setup()
{   
  pinMode(led2,OUTPUT);
      
  pinMode(led1,OUTPUT);
      
  pinMode(led3,OUTPUT);

   pinMode(led4,OUTPUT); 
      
 
}

void loop() 
{  digitalWrite(led3,LOW);
  digitalWrite(led4,LOW);
  while (count<255)
  {
   analogWrite(led2,count);
   analogWrite(led1,255-count);
   delay(10);
   count++;
  }
 while(count>0)
 {
 analogWrite(led2,count);
 analogWrite(led1,255-count);
 delay(10);
 count--;
 }
 
}
